194 int aws_main_build_certificate(t_aws_kit* kit)
195 {
196  int ret = AWS_E_FAILURE;
197  t_atcert cert;
198 
199  /* Allocate heap to obtain signer DER, PEM certificates and public key space. */
200  cert.signer_der = (uint8_t*)malloc(DER_CERT_INIT_SIZE);
201  if (cert.signer_der == NULL) goto free_cert;
202  cert.signer_der_size = DER_CERT_INIT_SIZE;
203  cert.signer_pem = (uint8_t*)malloc(PEM_CERT_INIT_SIZE);
204  if (cert.signer_pem == NULL) goto free_cert;
205  cert.signer_pem_size = PEM_CERT_INIT_SIZE;
206  cert.signer_pubkey= (uint8_t*)malloc(ATCERT_PUBKEY_SIZE);
207  if (cert.signer_pubkey == NULL) goto free_cert;
208 
209  /* Build signer certificate */
210  ret = atca_tls_build_signer_cert(&cert);
211  if (ret != ATCA_SUCCESS) {
212  ret = AWS_E_CRYPTO_CERT_FAILURE;
213  AWS_ERROR("Failed to build signer certificate!(%d)", ret);
214  goto free_cert;
215  }
216 
217  /* Allocate heap to secure device DER, PEM certificates and public key space. */
218  cert.device_der = (uint8_t*)malloc(DER_CERT_INIT_SIZE);
219  if (cert.device_der == NULL) goto free_cert;
220  cert.device_der_size = DER_CERT_INIT_SIZE;
221  cert.device_pem = (uint8_t*)malloc(PEM_CERT_INIT_SIZE);
222  if (cert.device_pem == NULL) goto free_cert;
223  cert.device_pem_size = PEM_CERT_INIT_SIZE;
224  cert.device_pubkey= (uint8_t*)malloc(ATCERT_PUBKEY_SIZE);
225  if (cert.device_pubkey == NULL) goto free_cert;
226 
227  /* Build device certificate. */
228  ret = atca_tls_build_device_cert(&cert);
229  if (ret != ATCA_SUCCESS) {
230  ret = AWS_E_CRYPTO_CERT_FAILURE;
231  AWS_ERROR("Failed to build device certificate!(%d)", ret);
232  goto free_cert;
233  }
234 
235  /* Copy both PEM certificates to corresponding fields to be used for TLS library. */
236  kit->cert.signerCertLen = cert.signer_pem_size;
237  memcpy(kit->cert.signerCert, cert.signer_pem, kit->cert.signerCertLen);
238  kit->cert.devCertLen = cert.device_pem_size;
239  memcpy(kit->cert.devCert, cert.device_pem, kit->cert.devCertLen);
240 
241  /* Release temporary heap region */
242 free_cert:
243  if (cert.signer_der) free(cert.signer_der);
244  if (cert.signer_pem) free(cert.signer_pem);
245  if (cert.signer_pubkey) free(cert.signer_pubkey);
246  if (cert.device_der) free(cert.device_der);
247  if (cert.device_pem) free(cert.device_pem);
248  if (cert.device_pubkey) free(cert.device_pubkey);
249 
250  return ret;
251 }

389 void aws_main_state_machine(t_aws_kit* kit)
390 {
391  int ret = AWS_E_SUCCESS;
392  static uint8_t currState = MAIN_STATE_INIT_KIT;
393  uint8_t nextState = MAIN_STATE_INVALID;
394 
395  switch (currState)
396  {
397  case MAIN_STATE_INIT_KIT:
398  /* Check if ATECC508A has been configured, and initialize BSP for ATWINC1500. */
399  ret = aws_main_init_kit(kit);
400  if (ret == AWS_E_CRYPTO_FAILURE) {
401  kit->errState = AWS_EX_UNPROVISIONED_CRYPTO;
402  aws_user_exception_init_timer(kit);
403  nextState = MAIN_STATE_PROVISIONING;
404  } else {
405  /* The kit has been initialized without any problem, go to next state */
406  nextState = MAIN_STATE_CHECK_KIT;
407  }
408  break;
409 
410  case MAIN_STATE_CHECK_KIT:
411  /* Check if all WIFI credential, AWS IoT host address and Thing name are available,
412  and build signer & device certificates. */
413  ret = aws_main_check_kit_state(kit);
414  if (ret != AWS_E_SUCCESS) {
415  kit->errState = AWS_EX_UNAVAILABLE_WIFI;
416  aws_user_exception_init_timer(kit);
417  nextState = MAIN_STATE_PROVISIONING;
418  } else {
419  kit->errState = AWS_EX_NONE;
420  nextState = MAIN_STATE_RUN_KIT;
421  kit->quitMQTT = false;
422  /* Resume Client task, if kit has been provisioned by using Insight GUI without any invalid configuration. */
423  vTaskResume(clientTaskHandler);
424  }
425  break;
426 
427  case MAIN_STATE_PROVISIONING:
428  /* Wait for a notification from Insight GUI to launch ,
429  if kit has been provisioned using Insight GUI without any invalid configuration. */
430  ret = aws_main_wait_notification(kit);
431  if (ret != AWS_E_SUCCESS) {
432  nextState = MAIN_STATE_PROVISIONING;
433  } else {
434  ret = aws_main_check_kit_state(kit);
435  if (ret != AWS_E_SUCCESS) {
436  nextState = MAIN_STATE_PROVISIONING;
437  } else {
438  kit->errState = AWS_EX_NONE;
439  nextState = MAIN_STATE_RUN_KIT;
440  kit->quitMQTT = false;
441  /* Resume Client task, if kit has been provisioned by using Insight GUI without any invalid configuration. */
442  vTaskResume(clientTaskHandler);
443  }
444  }
445  break;
446 
447  case MAIN_STATE_RUN_KIT:
448  /* Wait for a notification again to support multiple provisioning. */
449  ret = aws_main_wait_notification(kit);
450  if (ret == AWS_E_MQTT_REINITIALIZE) {
451  nextState = MAIN_STATE_PROVISIONING;
452  } else {
453  nextState = MAIN_STATE_RUN_KIT;
454  }
455  break;
456 
457  case MAIN_STATE_INVALID:
458  AWS_ERROR("Invalid main state");
459  nextState = MAIN_STATE_INVALID;
460  break;
461 
462  default:
463  break;
464  }
465 
466  currState = kit->mainState = nextState;
467 }
