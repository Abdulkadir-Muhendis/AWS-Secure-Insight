130 void aws_user_oled1_button_cb(uint32_t id, uint32_t mask)
131 {
132  t_aws_kit* kit = aws_kit_get_instance();
133  uint8_t buttonId = 0;
134  static uint8_t prevButtonId = 0;
135 
136  AWS_INFO("Pressed one of the buttons : %d %ld", kit->clientState, id);
137  if (kit->clientState != CLIENT_STATE_MQTT_WAIT_MESSAGE)
138  return;
139 
140  /* Set a pressed button. */
141  if ((OLED1_PIN_PUSHBUTTON_1_ID == id) && (OLED1_PIN_PUSHBUTTON_1_MASK == mask))
142  buttonId = OLED1_BUTTON1_ID;
143  else if ((OLED1_PIN_PUSHBUTTON_2_ID == id) && (OLED1_PIN_PUSHBUTTON_2_MASK == mask))
144  buttonId = OLED1_BUTTON2_ID;
145  else if ((OLED1_PIN_PUSHBUTTON_3_ID == id) && (OLED1_PIN_PUSHBUTTON_3_MASK == mask))
146  buttonId = OLED1_BUTTON3_ID;
147  else
148  return;
149 
150  /* Check for debounce issue. */
151  if (!(aws_kit_timer_expired(&kit->buttonISR)) && (buttonId == prevButtonId)) {
152  aws_kit_init_timer(&kit->buttonISR);
153  aws_kit_countdown_ms(&kit->buttonISR, 900);
154  return;
155  } else {
156  aws_kit_init_timer(&kit->buttonISR);
157  aws_kit_countdown_ms(&kit->buttonISR, 900);
158  }
159 
160  /* Queue the button ID pressed by a user. */
161  prevButtonId = buttonId;
162  if (!xQueueSendFromISR(kit->buttonQueue, (uint8_t *)&buttonId, NULL)) {
163  AWS_ERROR("Failed to send button ID!");
164  // Error: could not enqueue pressed button
165  }
166 }

255 void aws_user_exception_blink_led(t_aws_kit* kit)
256 {
257  if (kit->errState == AWS_EX_NONE || kit->errState == AWS_EX_MAX)
258  return;
259 
260  /* Don't blink, if the timer is not expired. */
261  if (!aws_kit_timer_expired(&kit->exceptionTimer))
262  return;
263 
264  /* Toggle IOPORT pin corresponding to exception state. */
265  AWS_ERROR("Exception happened!(%d)", kit->errState);
266  if (kit->errState == AWS_EX_UNPROVISIONED_CRYPTO) {
267  ioport_toggle_pin_level(OLED1_LED1_PIN);
268  } else if(kit->errState == AWS_EX_UNAVAILABLE_WIFI) {
269  ioport_toggle_pin_level(OLED1_LED2_PIN);
270  } else if(kit->errState == AWS_EX_MQTT_FAILURE) {
271  ioport_toggle_pin_level(OLED1_LED3_PIN);
272  } else {
273  return;
274  }
275 
276  /* Initialze a exception timer, and then count down the timer. */
277  aws_kit_init_timer(&kit->exceptionTimer);
278  aws_kit_countdown_sec(&kit->exceptionTimer, AWS_USER_ERROR_TIMEOUT_SEC);
279 }

286 void aws_user_task(void* params)
287 {
288  int ret = AWS_E_FAILURE;
289  t_aws_kit* kit = aws_kit_get_instance();
290  uint8_t butBuffer[1];
291 
292  kit->buttonQueue = xQueueCreate(1, sizeof(uint8_t));
293  kit->led.turn_on = aws_user_led_on;
294  kit->led.turn_off = aws_user_led_off;
295 
296  /* Initialize OLED1 board and SW0 button. */
297  aws_user_sw0_init();
298  aws_user_oled1_init();
299 
300  for (;;) {
301 
302  while (xQueueReceive(kit->buttonQueue, butBuffer, 0)) {
303  AWS_INFO("Pressed a button on the OLED1 : %d", butBuffer[0]);
304  if (butBuffer[0] == OLED1_BUTTON1_ID)
305  kit->button.isPressed[AWS_KIT_BUTTON_1] = AWS_BUTTON_PRESSED;
306  else if(butBuffer[0] == OLED1_BUTTON2_ID)
307  kit->button.isPressed[AWS_KIT_BUTTON_2] = AWS_BUTTON_PRESSED;
308  else if(butBuffer[0] == OLED1_BUTTON3_ID)
309  kit->button.isPressed[AWS_KIT_BUTTON_3] = AWS_BUTTON_PRESSED;
310  }
311 
312  /* If a user presses the SW0 button for 3 seconds, then reset all user data including WIFI credential, host address, etc. */
313  if (kit->pushButtonState == AWS_BUTTON_PRESSED) {
314  if (ioport_get_pin_level(BUTTON_0_PIN) == BUTTON_0_ACTIVE) {
315  if (aws_kit_timer_expired(&kit->resetISR)) {
316  /* Turn on all LEDs to notify reboot. */
317  kit->pushButtonState = AWS_BUTTON_RELEASED;
318  ioport_set_pin_level(LED_0_PIN, LED_0_INACTIVE);
319  ioport_set_pin_level(OLED1_LED1_PIN, LED_0_ACTIVE);
320  ioport_set_pin_level(OLED1_LED2_PIN, LED_0_ACTIVE);
321  ioport_set_pin_level(OLED1_LED3_PIN, LED_0_ACTIVE);
322  aws_kit_init_timer(&kit->resetISR);
323  ret = aws_main_reset_user_data(kit);
324  if (ret != ATCA_SUCCESS) {
325  AWS_ERROR("Failed to reset user data!(%d)", ret);
326  ret = AWS_E_CRYPTO_FAILURE;
327  break;
328  }
329  /* Reboot the kit. */
330  aws_kit_software_reset();
331  }
332  } else {
333  kit->pushButtonState = AWS_BUTTON_RELEASED;
334  ioport_set_pin_level(LED_0_PIN, LED_0_INACTIVE);
335  aws_kit_init_timer(&kit->resetISR);
336  }
337  }
338 
339  /* Once a exception happens, then blink the specified LED. */
340  if ((kit->errState != AWS_EX_NONE) && (kit->errState > AWS_EX_NONE && kit->errState < AWS_EX_MAX)) {
341  aws_user_exception_blink_led(kit);
342  }
343 
344  /* Block for 100ms. */
345  vTaskDelay(AWS_USER_TASK_DELAY);
346  }
347 }
