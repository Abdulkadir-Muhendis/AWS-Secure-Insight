175 {

178  t_aws_kit* kit = aws_kit_get_instance();

181 
182 #ifdef AWS_KIT_DEBUG
183  wolfSSL_Debugging_ON();
184 #else
185  wolfSSL_Debugging_OFF();
186  delay_ms(500);
187 #endif
188  /* Initialize SSL context. */
189  mqttCli->tls.ctx = wolfSSL_CTX_new(wolfTLSv1_2_client_method());
190  if (mqttCli->tls.ctx == NULL) {
191  AWS_ERROR("Failed to init context!");
192  break;
193  }
194 
195  /* ATECC508A supports ECC(ECDH, ECDSA) hardware acceleration.
196  ECDHE-ECDSA-AES128-GCM-SHA256 cipher suite recommende by AWS IoT is set for authentication and data encryption. */
197  ret = wolfSSL_CTX_set_cipher_list(mqttCli->tls.ctx, AWS_IOT_CIPHER_SPEC);
198  if (ret != SSL_SUCCESS) {
199  AWS_ERROR("Failed to set cipher!");
200  break;
201  }
202 
203  /* Since AWS IoT server was signed by the VeriSign root CA, this root CA certificate should be loaded to WolfSSL to verify AWS ioT. */
204  ret = wolfSSL_CTX_load_verify_buffer(mqttCli->tls.ctx, AWS_IOT_ROOT_CERT, sizeof(AWS_IOT_ROOT_CERT), SSL_FILETYPE_PEM);
205  if (ret != SSL_SUCCESS) {
206  AWS_ERROR("Failed to set root cert!");
207  break;
208  }
209 
210  /* As the AT88CKECCSIGNER already signed ATECC508A of Thing, There are the Signer and Device certifictes in the ATECC508A.
211  Both certificates should be set to WolfSSL for the JITR achievement. */
212  cert_chain = (uint8_t*)malloc(kit->cert.signerCertLen + kit->cert.devCertLen);
213  memcpy(&cert_chain[0], kit->cert.devCert, kit->cert.devCertLen);
214  memcpy(&cert_chain[kit->cert.devCertLen], kit->cert.signerCert, kit->cert.signerCertLen);
215  ret = wolfSSL_CTX_use_certificate_chain_buffer(mqttCli->tls.ctx, cert_chain, kit->cert.signerCertLen + kit->cert.devCertLen);
216  if (ret != SSL_SUCCESS) {
217  AWS_ERROR("Failed to set cert chain!");
218  break;
219  }
220 
221  /* This Device private key is not actual private key of ATECC508A, and WolfSSL never use it as own Device key.
222  This temporaty key has been set to make sure Device owns a private key. */
223  ret = wolfSSL_CTX_use_PrivateKey_buffer(mqttCli->tls.ctx, AWS_TEMP_DEV_KEY, sizeof(AWS_TEMP_DEV_KEY), SSL_FILETYPE_PEM);
224  if (ret != SSL_SUCCESS) {
225  AWS_ERROR("Failed to set fake key!");
226  break;
227  }
228 
229  /* Turn on a certificate request from the server to the client. */
230  wolfSSL_CTX_set_verify(mqttCli->tls.ctx, SSL_VERIFY_PEER, NULL);
231  /* Set the Public key Callback for ECC Signing. */
232  wolfSSL_CTX_SetEccSignCb(mqttCli->tls.ctx, atca_tls_sign_certificate_cb);
233  /* Set the Public key Callback for ECC Verification. */
234  wolfSSL_CTX_SetEccVerifyCb(mqttCli->tls.ctx, atca_tls_verify_signature_cb);
235  /* Set the Public key Callback for Pre-Master Secret creation. */
236  wolfSSL_CTX_SetEccPmsCb(mqttCli->tls.ctx, atca_tls_create_pms_cb);

238  } while(0);

561 void aws_client_state_machine(t_aws_kit* kit)
562 {
563  int ret = AWS_E_SUCCESS;
564  static bool errorNoti = false;
565  static uint8_t retryDelay = 0;
566  static uint8_t currState = CLIENT_STATE_INIT_MQTT_CLIENT;
567  uint8_t nextState = CLIENT_STATE_INVALID;
568 
569  switch (currState)
570  {
571  case CLIENT_STATE_INIT_MQTT_CLIENT:
572  {
573  /* After initializing WolfMQTT, try to connect to host address. */
574  ret = aws_client_init_mqtt_client(kit);
575  if (ret == AWS_E_SUCCESS) {
576  retryDelay = 0;
577  kit->errState = AWS_EX_NONE;
578  errorNoti = false;
579  for (uint8_t i = 0; i < AWS_KIT_LED_MAX; i++) {
580  kit->led.turn_off((uint8_t)i + 1);
581  }
582  /* Since Thing is ready to communicate securely with AWS IoT, go to next state. */
583  nextState = CLIENT_STATE_MQTT_SUBSCRIBE;
584  /* For the Just In Time Regstration, hold on seconds, and retry */
585  } else if (ret == AWS_E_NET_JITR_RETRY || AWS_E_NET_TLS_FAILURE) {
586  retryDelay += 2;
587  if (retryDelay > 120) {
588  AWS_ERROR("Failed to exceed the limited 2 minutes to wait for finishing lambda");
589  aws_kit_software_reset();
590  }
591  delay_s((unsigned long)retryDelay);
592  nextState = CLIENT_STATE_INIT_MQTT_CLIENT;
593  } else {
594  if (!errorNoti) {
595  errorNoti = true;
596  kit->errState = AWS_EX_MQTT_FAILURE;
597  aws_user_exception_init_timer(kit);
598  }
599  nextState = CLIENT_STATE_INIT_MQTT_CLIENT;
600  }
601  }
602  break;
603 
604  case CLIENT_STATE_MQTT_SUBSCRIBE:
605  {
606  /* Subscribe the Delta topic. */
607  ret = aws_client_mqtt_subscribe(kit);
608  if (ret != AWS_E_SUCCESS) {
609  if (!errorNoti) {
610  errorNoti = true;
611  kit->errState = AWS_EX_MQTT_FAILURE;
612  aws_user_exception_init_timer(kit);
613  }
614  nextState = CLIENT_STATE_MQTT_SUBSCRIBE;
615  } else {
616  kit->errState = AWS_EX_NONE;
617  errorNoti = false;
618  nextState = CLIENT_STATE_MQTT_PUBLISH;
619  }
620  }
621  break;
622 
623  case CLIENT_STATE_MQTT_PUBLISH:
624  {
625  /* Publish the Update topic. */
626  ret = aws_client_mqtt_publish(kit);
627  if (ret != AWS_E_SUCCESS) {
628  if (!errorNoti) {
629  errorNoti = true;
630  kit->errState = AWS_EX_MQTT_FAILURE;
631  aws_user_exception_init_timer(kit);
632  }
633  nextState = CLIENT_STATE_MQTT_PUBLISH;
634  } else {
635  kit->errState = AWS_EX_NONE;
636  errorNoti = false;
637  nextState = CLIENT_STATE_MQTT_WAIT_MESSAGE;
638  }
639  }
640  break;
641 
642  case CLIENT_STATE_MQTT_WAIT_MESSAGE:
643  {
644  if (kit->quitMQTT) {
645  ret = aws_client_mqtt_disconnect(kit);
646  if (ret != AWS_E_SUCCESS) {
647  if (!errorNoti) {
648  errorNoti = true;
649  kit->errState = AWS_EX_MQTT_FAILURE;
650  aws_user_exception_init_timer(kit);

653  } else {

671  }
672  }

674

679  }
680  break;
681

683  break;
684  }
685
